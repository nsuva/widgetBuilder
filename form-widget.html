Simply paste this code into your sites HTML where you'd like the button to appear (don't forget to set the amount, description values and set what you want to do on payment success and failure)...

<!-- Load core functionality //-->
<script src="://mattpass.com/lab/widget.js" id="btcpWidget"></script>

<!-- Set parameters and actions //-->
<script>
  btcpWidget.data = {
    "wallet"      : "walletaddress1234567890",
    "amount"      : "12.3456",
    "description" : "Shoes"
  };

  btcpWidget.onPaymentSuccess = function(data) {
    alert("Payment success! " + data.token);
  };

  btcpWidget.onPaymentFail = function(data) {
    alert("Payment failed! " + data.reason);
  }
</script>
